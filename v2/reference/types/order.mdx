## Order

The `Order` interface represents a customer order.

```typescript
import type { Order } from '@shopperlabs/shopper-types'
```

### Definition

```typescript
interface Order extends Entity {
  number: string
  price_amount: number
  notes: string | null
  currency_code: string
  total_amount?: number
  status: OrderStatus
  canceled_at: DateEntity | null
  archived_at: DateEntity | null
  zone_id: number | null
  shipping_address_id: number | null
  billing_address_id: number | null
  payment_method_id: number | null
  customer_id: number | null
  channel_id: number | null
  parent_order_id: number | null
  shipping_option_id?: number | null

  // Relationships
  shippingOption?: CarrierOption
  shippingAddress?: OrderAddress | null
  billingAddress?: OrderAddress | null
  paymentMethod?: PaymentMethod | null
  zone?: Zone | null
  channel?: Channel | null
  parent?: Order | null
  customer?: Customer
  items?: OrderItem[]
  shippings?: OrderShipping[]
  children?: Order[]
  refund?: OrderRefund
}
```

### OrderStatus Enum

```typescript
enum OrderStatus {
  PENDING = 'pending',
  NEW = 'new',
  REGISTER = 'register',
  PAID = 'paid',
  COMPLETED = 'completed',
  SHIPPED = 'shipped',
  CANCELLED = 'cancelled',
}
```

### Usage Examples

```typescript
import type { Order } from '@shopperlabs/shopper-types'
import { OrderStatus } from '@shopperlabs/shopper-types'

// Check if order can be cancelled
function canCancel(order: Order): boolean {
  return [OrderStatus.PENDING, OrderStatus.NEW].includes(order.status)
}

// Calculate order total
function calculateTotal(order: Order): number {
  return order.items?.reduce((sum, item) => sum + item.total, 0) ?? 0
}

// Get status badge color
function getStatusColor(status: OrderStatus): string {
  const colors: Record<OrderStatus, string> = {
    [OrderStatus.PENDING]: '#fbbf24',
    [OrderStatus.NEW]: '#3b82f6',
    [OrderStatus.REGISTER]: '#6b7280',
    [OrderStatus.PAID]: '#10b981',
    [OrderStatus.COMPLETED]: '#059669',
    [OrderStatus.SHIPPED]: '#8b5cf6',
    [OrderStatus.CANCELLED]: '#ef4444',
  }
  return colors[status]
}
```
